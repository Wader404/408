int visited[MVNum];
int level = 1;
int PathDFS(ALGraph G, int i, int j)
{
    if (i == j) {
        return 1;
    } else {
        visited[i] = true;
        for (p = G.vertices[i].firstarc; p; p = p -> nextarc, level--) {
            level++;
            k = p -> adjvex;
            if (!visited[k] && PathDFS(G, k, j)) {
                return 1;
            }
        }
    }
    if (level == 1) return 0;
}
